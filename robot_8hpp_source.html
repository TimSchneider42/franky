488 template <class Rep = long, class Period = std::ratio<1>>
489 bool joinMotionUnsafe(
490 std::unique_lock<std::mutex> &lock,
491 const std::optional<std::chrono::duration<Rep, Period>> &timeout = std::nullopt) {
492 while (motion_generator_running_) {
493 if (timeout.has_value()) {
494 if (control_finished_condition_.wait_for(lock, timeout.value()) == std::cv_status::timeout) {
495 return false;
496 }
497 } else {
498 control_finished_condition_.wait(lock);
499 }
500 }
501 if (control_thread_.joinable()) control_thread_.join();
502 if (control_exception_ != nullptr) {
503 auto control_exception = control_exception_;
504 control_exception_ = nullptr;
505 std::rethrow_exception(control_exception);
506 }
507 return true;
508 }

510 template <typename ControlSignalType>
511 void moveInternal(
512 const std::shared_ptr<Motion<ControlSignalType>> &motion,
513 const std::function<void(const ControlFunc<ControlSignalType> &)> &control_func_executor, bool async) {
514 if (motion == nullptr) {
515 throw std::invalid_argument("The motion must not be null.");
516 }
517 { // Do not remove brace, it is needed to scope the lock
518 std::unique_lock lock(*control_mutex_);
519 if (is_in_control_unsafe() && motion_generator_running_) {
520 if (!std::holds_alternative<MotionGenerator<ControlSignalType>>(motion_generator_)) {
521 throw InvalidMotionTypeException(
522 "The type of motion cannot change during runtime. Please ensure that the "
523 "previous motion finished before using a new type of motion.");
524 }
525 std::get<MotionGenerator<ControlSignalType>>(motion_generator_).updateMotion(motion);
526 } else {
527 joinMotionUnsafe(lock);
528
529 motion_generator_.emplace<MotionGenerator<ControlSignalType>>(this, motion);
530 auto motion_generator = &std::get<MotionGenerator<ControlSignalType>>(motion_generator_);
531 motion_generator->registerUpdateCallback(
532 [this](const RobotState &robot_state, franka::Duration duration, franka::Duration time) {
533 std::lock_guard lock(state_mutex_);
534 current_state_ = robot_state;
535 });
536 motion_generator_running_ = true;
537 control_thread_ = std::thread([this, control_func_executor, motion_generator]() {
538 try {
539 bool done = false;
540 RobotStateEstimator robot_state_estimator(
541 params_.dq_estimator_window_size,
542 params_.ddq_estimator_window_size,
543 params_.dq_estimator_alpha,
544 params_.ddq_estimator_alpha);
545 while (!done) {
546 control_func_executor(
547 [this, motion_generator, &robot_state_estimator](const franka::RobotState &rs, franka::Duration d) {
548 return (*motion_generator)(robot_state_estimator.update(rs, *model_), d);
549 });
550 std::unique_lock lock(*control_mutex_);
551
552 // This code is just for the case that a new motion is set just after the old one terminates. If this
553 // happens, we need to continue with this motion, unless an exception occurs.
554 done = !motion_generator->has_new_motion();
555 if (motion_generator->has_new_motion()) {
556 motion_generator->resetTimeUnsafe();
557 } else {
558 done = true;
559 motion_generator_running_ = false;
560 control_finished_condition_.notify_all();
561 }
562 }
563 } catch (...) {
564 std::unique_lock lock(*control_mutex_);
565 control_exception_ = std::current_exception();
566 motion_generator_running_ = false;
567 control_finished_condition_.notify_all();
568 }
569 });
570 }
571 }
572 if (!async) joinMotion();
573 }
