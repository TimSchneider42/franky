479 template<class Rep = long, class Period = std::ratio<1>>
480 bool joinMotionUnsafe(
481 std::unique_lock<std::mutex> &lock,
482 const std::optional<std::chrono::duration<Rep, Period>> &timeout = std::nullopt) {
483 while (motion_generator_running_) {
484 if (timeout.has_value()) {
485 if (control_finished_condition_.wait_for(lock, timeout.value()) == std::cv_status::timeout) {
486 return false;
487 }
488 } else {
489 control_finished_condition_.wait(lock);
490 }
491 }
492 if (control_thread_.joinable())
493 control_thread_.join();
494 if (control_exception_ != nullptr) {
495 auto control_exception = control_exception_;
496 control_exception_ = nullptr;
497 std::rethrow_exception(control_exception);
498 }
499 return true;
500 }

502 template<typename ControlSignalType>
503 void moveInternal(
504 const std::shared_ptr<Motion<ControlSignalType>> &motion,
505 const std::function<void(const ControlFunc<ControlSignalType> &)> &control_func_executor,
506 bool async) {
507 if (motion == nullptr) {
508 throw std::invalid_argument("The motion must not be null.");
509 }
510 std::unique_lock lock(*control_mutex_);
511 if (is_in_control_unsafe() && motion_generator_running_) {
512 if (!std::holds_alternative<MotionGenerator<ControlSignalType>>(motion_generator_)) {
513 throw InvalidMotionTypeException("The type of motion cannot change during runtime. Please ensure that the "
514 "previous motion finished before using a new type of motion.");
515 } else {
516 std::get<MotionGenerator<ControlSignalType>>(motion_generator_).updateMotion(motion);
517 }
518 } else {
519 joinMotionUnsafe(lock);
520
521 motion_generator_.emplace<MotionGenerator<ControlSignalType>>(this, motion);
522 auto motion_generator = &std::get<MotionGenerator<ControlSignalType>>(motion_generator_);
523 motion_generator->registerUpdateCallback(
524 [this](const franka::RobotState &robot_state,
525 franka::Duration duration,
526 franka::Duration time) {
527 std::lock_guard lock(*this->state_mutex_);
528 current_state_ = robot_state;
529 });
530 motion_generator_running_ = true;
531 control_thread_ = std::thread(
532 [this, control_func_executor, motion_generator]() {
533 try {
534 bool done = false;
535 while (!done) {
536 control_func_executor(
537 [motion_generator](const franka::RobotState &rs, franka::Duration d) {
538 return (*motion_generator)(rs, d);
539 });
540 std::unique_lock lock(*control_mutex_);
541
542 // This code is just for the case that a new motion is set just after the old one terminates. If this
543 // happens, we need to continue with this motion, unless an exception occurs.
544 done = !motion_generator->has_new_motion();
545 if (motion_generator->has_new_motion()) {
546 motion_generator->resetTimeUnsafe();
547 } else {
548 done = true;
549 motion_generator_running_ = false;
550 control_finished_condition_.notify_all();
551 }
552 }
553 } catch (...) {
554 std::unique_lock lock(*control_mutex_);
555 control_exception_ = std::current_exception();
556 motion_generator_running_ = false;
557 control_finished_condition_.notify_all();
558 }
559 }
560 );
561 }
562 if (!async)
563 joinMotion();
564 }
