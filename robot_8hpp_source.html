457 template<class Rep = long, class Period = std::ratio<1>>
458 bool joinMotionUnsafe(
459 std::unique_lock<std::mutex> &lock,
460 const std::optional<std::chrono::duration<Rep, Period>> &timeout = std::nullopt) {
461 while (motion_generator_running_) {
462 if (timeout.has_value()) {
463 if (control_finished_condition_.wait_for(lock, timeout.value()) == std::cv_status::timeout) {
464 return false;
465 }
466 } else {
467 control_finished_condition_.wait(lock);
468 }
469 }
470 if (control_thread_.joinable())
471 control_thread_.join();
472 if (control_exception_ != nullptr) {
473 auto control_exception = control_exception_;
474 control_exception_ = nullptr;
475 std::rethrow_exception(control_exception);
476 }
477 return true;
478 }

480 template<typename ControlSignalType>
481 void moveInternal(
482 const std::shared_ptr<Motion<ControlSignalType>> &motion,
483 const std::function<void(const ControlFunc<ControlSignalType> &)> &control_func_executor,
484 bool async) {
485 if (motion == nullptr) {
486 throw std::invalid_argument("The motion must not be null.");
487 }
488 { // Do not remove brace, it is needed to scope the lock
489 std::unique_lock lock(*control_mutex_);
490 if (is_in_control_unsafe() && motion_generator_running_) {
491 if (!std::holds_alternative<MotionGenerator<ControlSignalType>>(motion_generator_)) {
492 throw InvalidMotionTypeException("The type of motion cannot change during runtime. Please ensure that the "
493 "previous motion finished before using a new type of motion.");
494 } else {
495 std::get<MotionGenerator<ControlSignalType>>(motion_generator_).updateMotion(motion);
496 }
497 } else {
498 joinMotionUnsafe(lock);
499
500 motion_generator_.emplace<MotionGenerator<ControlSignalType>>(this, motion);
501 auto motion_generator = &std::get<MotionGenerator<ControlSignalType>>(motion_generator_);
502 motion_generator->registerUpdateCallback(
503 [this](const RobotState &robot_state,
504 franka::Duration duration,
505 franka::Duration time) {
506 std::lock_guard lock(state_mutex_);
507 current_state_ = robot_state;
508 });
509 motion_generator_running_ = true;
510 control_thread_ = std::thread(
511 [this, control_func_executor, motion_generator]() {
512 try {
513 bool done = false;
514 while (!done) {
515 control_func_executor(
516 [this, motion_generator](const franka::RobotState &rs, franka::Duration d) {
517 return (*motion_generator)(RobotState::from_franka(rs), d);
518 });
519 std::unique_lock lock(*control_mutex_);
520
521 // This code is just for the case that a new motion is set just after the old one terminates. If this
522 // happens, we need to continue with this motion, unless an exception occurs.
523 done = !motion_generator->has_new_motion();
524 if (motion_generator->has_new_motion()) {
525 motion_generator->resetTimeUnsafe();
526 } else {
527 done = true;
528 motion_generator_running_ = false;
529 control_finished_condition_.notify_all();
530 }
531 }
532 } catch (...) {
533 std::unique_lock lock(*control_mutex_);
534 control_exception_ = std::current_exception();
535 motion_generator_running_ = false;
536 control_finished_condition_.notify_all();
537 }
538 }
539 );
540 }
541 }
542 if (!async)
543 joinMotion();
544 }
